# Single-process NiceGUI app + sklearn pickle compatibility.
#
# model_pipeline.pkl targets scikit-learn 1.6.x — keep <1.7 for pickle safety.
#
# NumPy 2.x + Anaconda: conda’s optional pandas deps (numexpr, bottleneck) are often still
# built against NumPy 1.x → "AttributeError: _ARRAY_API not found" / "compiled using NumPy 1.x".
# Pin NumPy 1.26 on Python <3.13 for a stable ABI with pandas 2.2, scipy, sklearn, and LightGBM.
#
# Python 3.13+: NumPy 1.26 has no manylinux wheels → pip builds from source and needs gcc.
# Use NumPy 2.1+ there (prebuilt wheels). pandas 2.2.3 + scikit-learn 1.6.x are compatible.
nicegui>=2.0.0
pandas==2.2.3
numpy==1.26.4; python_version < "3.13"
numpy>=2.1.3,<2.3; python_version >= "3.13"
# SciPy 1.14+ aligns with NumPy 2 when the resolver picks Python 3.13 + NumPy 2.x.
scipy>=1.14.1,<1.15
scikit-learn==1.6.1
lightgbm==4.5.0
joblib==1.4.2
plotly>=5.18.0
openpyxl>=3.1.0
